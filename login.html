function register(){ let username = document.getElementById("username").value.trim(); let password = document.getElementById("password").value.trim(); if(!username || !password){ alert("نام کاربری و رمز را وارد کنید"); return; } let users = JSON.parse(localStorage.getItem("users")) || []; let check = users.find( u => u.username === username ); if(check){ alert("این نام کاربری قبلا ثبت شده"); return; } let user = { id:Date.now(), username:username, password:password, coins:100000, wins:0, games:0, record:0, history:[] }; users.push(user); localStorage.setItem( "users", JSON.stringify(users) ); alert("ثبت نام موفق بود"); location.href="login.html"; }
